I made this wepsite using Angular 8 and Docker.

# stage 1
FROM node:latest as node
WORKDIR /app
COPY . .
RUN npm install
RUN npm run build —- —-prod


# stage 2
FROM nginx:alpine
COPY --from=node /app/dist/cagri.tk /usr/share/nginx/html
